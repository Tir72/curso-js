Primitivos

  • string
  • number
  • boolean
  • null
  • undefined

string

Cadenas de texto de cero o más caracteres contenidos entre comillas dobles o sencillas

var myString = "this class is awesome";
log(myString);

number

Pueden ser enteros o flotantes, siempre serán tratados como números

var myNumber = 56;
log(myNumber);

boolean

Los valores booleanos solo pueden tener dos valores: falso o verdadero 

var myBooleanAmaya = true;
log(myBooleanAmaya);

null

Un valor null es un valor indefinido que es asignado intencionalmente

var myNullVal = null;
log(myNullVal);

undefined

El valor "undefined" es asignado a todas las variables que no fueron inicializadas, y es retornado también al usar variables no declaradas

var myUndefined;
log(myUndefined);

Object

  • Es una colección de propiedades a las cuales les puedo asignar un valor.
  • Las propiedades son cadenas de texto
  • Los valores pueden ser primitivos u otros objetos
  • Pueden ser construidos usando {} o directamente el constructor Object
  • Todo objeto tiene un valor primitivo y/o un valor string implícito

Object

  • Es una colección de propiedades a las cuales les puedo asignar un valor.
  • Las propiedades son cadenas de texto
  • Los valores pueden ser primitivos u otros objetos
  • Pueden ser construidos usando {} o directamente el constructor Object
  • Todo objeto tiene un valor primitivo y/o un valor string implícito

Object

var obj = {}; // objeto vacío
var obj2 = {
  name: 'Johnny Bravo', // string
  age: 23, // number
  active: false // boolean
};

Object

var obj = new Object(); // objeto vacío, equivalente a {}

var obj2 = new Object();

obj2.name = 'Johnny Bravo'; // string
obj2.age = 23; // number
obj2.active = false; // boolean

Nested Object

var obj3 = {
  codename: 'JohnnyB', // string
  data: { // object
    realName: 'Johnny Bravo',
    age: 23,
    active: false
  }  
};

Nested Object

obj3.codename;
obj3.data.realName;
obj3.data.age;

indexOf

var test = 'Lorem ipsum dolor sit amet'
test.indexOf('Lorem') // 0
test.indexOf('d') // 12
test.indexOf('vestibulum') // -1
test.indexOf('lorem') // -1 sensiblea mayúsculas
test.indexOf(/lorem/i) // 0 regexp

lastIndexOf

var test = 'Lorem ipsum dolor sit amet'
test.lastIndexOf('amet') // 22
test.lastIndexOf('m') // 23
test.lastIndexOf('vestibulum') // -1
test.lastIndexOf(/AMET/i) // 22 regexp

match

var test = 'Lorem ipsum dolor sit amet lorem'
test.match('amet') // ['amet']
test.match('lorem') // ['lorem']
test.match(/lorem/i) // ['Lorem'] regexp
test.match(/lorem/ig) // ['Lorem', 'lorem']

toUpperCase / toLowerCase

var test = 'Lorem ipsum dolor sit AMET'
test.toUpperCase() // 'LOREM IPSUM DOLOR SIT AMET'
test.toLowerCase() // 'lorem ipsum dolor sit amet'

substring / substr

var test = 'Lorem ipsum dolor sit amet'
test.substring(0,5) // 'Lorem'
test.substring(11,6) // 'ipsum'
test.substring(11,-90) // 'Lorem ipsum'
test.substring(0,0) // ''
test.substr(0,5) // 'Lorem'
test.substr(11,6) // ' dolor'
test.substr(11,-1) // ''
test.substr(0,0) // ''

split

var test = 'Lorem ipsum dolor sit amet'
test.split('m') // ['Lore', ' ipsu', ' dolor sit a', 'et']
test.split('ipsum') // ['Lorem ', ' dolor sit amet']

test = 'lunes,martes,miércoles,jueves,viernes'
test.split(',') // ['lunes', 'martes', 'miércoles', 'jueves', 'viernes']

trim

var test = ' hola     '
test.trim() // 'hola'

test = ' hola      mundo  '
test.trim() // 'hola      mundo'
